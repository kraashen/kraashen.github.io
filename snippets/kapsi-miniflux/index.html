RewriteEngine On

RewriteRule ^index\.html$ http://webapp1.n.kapsi.fi:port/ [P] 
RewriteRule ^(.*)$ http://webapp1.n.kapsi.fi:port/$1 [P]

run-miniflux.sh
---

#!/usr/bin/env bash

flock \
  --exclusive \
  --nonblock \
  --conflict-exit-code 0 \
  ~/.lock/miniflux.lock \
  ~/sites/nick.kapsi.fi/miniflux/miniflux -c ~/.local/etc/miniflux/miniflux.conf
